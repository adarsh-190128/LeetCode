class Solution {
public:
    vector<int> runningSum(vector<int>& nums) {
        // initiliaze a variable to store sum of elements 
        
        int sum=0;
        
        // now initialize a loop to iterate the array and sum
        
        for(int i=0;i<nums.size();i++){
            sum+=nums[i];
            nums[i]=sum;
        }
        return nums;
    }
};
